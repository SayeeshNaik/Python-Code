var nextButton = document.querySelector('.nextbtn');
nextButton.addEventListener('click',function good(){
    fetch('http://127.0.0.1:5000/next_button_sending').then(res=>res.json().then(
        resData=>{
            var next_button_xpath = resData.next_button_xpath;
            if(resData.next_button_status==true){
                var next_button_el = document.querySelector('.nextbtn');
                next_button_el.style.background = 'green';
                next_button_el.style.color = 'white'
                var next_button_pencil = document.getElementById('xpath-pencil');
                next_button_pencil.style.color = 'blue';
                var next_button_xpath_txt = document.getElementById('nextBtnXpath');
                next_button_xpath_txt.value = next_button_xpath;
            }else{
                var next_button_el = document.querySelector('.nextbtn');
                next_button_el.style.background = 'red';
                next_button_el.style.color = 'black'
                var next_button_pencil = document.getElementById('xpath-pencil');
                next_button_pencil.style.color = 'black';

            }
        }
    ))
})
